// Algorithm to solve fatCaloriesPerServing

Begin

    Declare "foodName" as string
    Declare Real fatGrams
    Declare Real numberOfServings
    Declare Integer caloriesPerFatGram 
    Declare Real totalFatCalories
    Declare Real fatCaloriesPerServing

    const caloriesPerFatGram = 9

    Display "Enter name of food, fatgrams, and number of servings:"
    Input foodName
    Input fatGrams
    Input numberOfServings

    totalFatCalories = fatGrams * caloriesPerFatGram

    fatCaloriesPerServing = totalFatCalories / numberOfServings
    return fatCaloriesPerServing

    Display "FatCalories in "foodName" per serving:"

End

BEGIN

    DECLARE foodName AS STRING
    DECLARE fatGrams AS REAL
    DECLARE servings AS INTEGER
    DECLARE caloriesPerGram AS INTEGER
    DECLARE totalCalories AS REAL
    DECLARE caloriesPerServing AS REAL

    SET caloriesPerGram = 9

    DISPLAY "Enter name of food item, fatgrams and number of servings:"
    INPUT foodName
    INPUT fatGrams
    INPUT servings

    totalCalories = fatGrams * caloriesPerGram

    caloriesPerServing = totalCalories / servings

    DISPLAY "Calories in ", foodName, " per serving: ", caloriesPerServing

END
